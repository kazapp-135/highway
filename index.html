// 📍 本物のGPS 3点計測＆ラズパイへの解析リクエスト async function startGPSAnalysis() { const status = document.getElementById('status-display'); const btn = document.getElementById('gps-btn'); document.getElementById('timeline-container').innerHTML = ""; btn.disabled = true; let capturedCoords = []; // 位置情報を取得するPromise関数 const getPos = () => new Promise((resolve, reject) => { navigator.geolocation.getCurrentPosition(resolve, reject, { enableHighAccuracy: true, timeout: 5000, maximumAge: 0 }); }); try { // 1秒に1回、計3回のGPS座標を取得 for (let i = 1; i <= 3; i++) { status.innerText = `[${i}/3] TRACKING SATELLITE...`; const pos = await getPos(); capturedCoords.push({ lat: pos.coords.latitude, lon: pos.coords.longitude }); if (i < 3) await new Promise(r => setTimeout(r, 1000)); } status.innerText = "DATA ACQUIRED. CALCULATING VECTOR..."; // ラズパイ（頭脳）へ座標データを投げて解析させる const res = await fetch(`${API_BASE_URL}/api/analyze_route`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ coords: capturedCoords }) }); const result = await res.json(); if (result.status === "success") { status.innerText = `TARGET ACQUIRED: 誤差 ${result.distance_m}m で捕捉`; renderTimeline(result.highway, "timeline-container"); } else { status.innerText = "ERROR: " + result.message; status.style.color = "var(--alert-red)"; } } catch (error) { status.innerText = "ERROR: GPS信号を受信できません。"; status.style.color = "var(--alert-red)"; } finally { btn.disabled = false; btn.innerText = "POSITION RE-SCAN"; } }
